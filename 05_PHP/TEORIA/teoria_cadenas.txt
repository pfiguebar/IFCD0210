*****  strlen(variable)  *****
<?php
$cad = "cadena";  
$numero = strlen($cad);                            // devuelve la longitud de la cadena  --> en este caso 6
$agua = "litros"

echo "La longitud de la cadena es:"  . $numero  .$agua    // concatenados - esto imprime en pantalla -->  La longitud de la cadena es: 6 litros
echo "La longitud de la cadena es: $numero $agua"        // al ir dentro " ..... imprime texto interpretado --> La longitud de la cadena es: 6 litros

echo 'La longitud de la cadena es: $numero'             // al ir dentro ' ..... imprime texto literal --> La longitud de la cadena es: $numero
?>

*****  count_chars(variable, 1)  *****
<?php
$cadena = "es jueves";
$mi_array = count_chars($cadena, 1);

echo $mi_array     // imprime e(3 veces)  s(2 veces)  j(1 vez)  u(1 vez) .....
?>



*****  substr.count($variable, $cadena)  *****
<?php
